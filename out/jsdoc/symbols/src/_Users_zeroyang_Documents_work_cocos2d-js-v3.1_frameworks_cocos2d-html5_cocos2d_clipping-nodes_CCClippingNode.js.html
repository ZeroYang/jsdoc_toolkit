194     _visitForWebGL: function (ctx) {
195         var gl = ctx || cc._renderContext;
196 
197         // if stencil buffer disabled
198         if (cc.stencilBits < 1) {
199             // draw everything, as if there where no stencil
200             cc.Node.prototype.visit.call(this, ctx);
201             return;
202         }
203 
204         if (!this._stencil || !this._stencil.visible) {
205             if (this.inverted)
206                 cc.Node.prototype.visit.call(this, ctx);   // draw everything
207             return;
208         }
209 
210         if (cc.ClippingNode._layer + 1 == cc.stencilBits) {
211             cc.ClippingNode._visit_once = true;
212             if (cc.ClippingNode._visit_once) {
213                 cc.log("Nesting more than " + cc.stencilBits + "stencils is not supported. Everything will be drawn without stencil for this node and its childs.");
214                 cc.ClippingNode._visit_once = false;
215             }
216             // draw everything, as if there where no stencil
217             cc.Node.prototype.visit.call(this, ctx);
218             return;
219         }
220 
221         cc.renderer.pushRenderCommand(this._beforeVisitCmd);
222 
223         //optimize performance for javascript
224         var currentStack = cc.current_stack;
225         currentStack.stack.push(currentStack.top);
226         cc.kmMat4Assign(this._stackMatrix, currentStack.top);
227         currentStack.top = this._stackMatrix;
228 
229         this.transform();
230         //this._stencil._stackMatrix = this._stackMatrix;
231         this._stencil.visit();
232 
233         cc.renderer.pushRenderCommand(this._afterDrawStencilCmd);
234 
235         // draw (according to the stencil test func) this node and its childs
236         var locChildren = this._children;
237         if (locChildren && locChildren.length > 0) {
238             var childLen = locChildren.length;
239             this.sortAllChildren();
240             // draw children zOrder < 0
241             for (var i = 0; i < childLen; i++) {
242                 if (locChildren[i] && locChildren[i]._localZOrder < 0)
243                     locChildren[i].visit();
244                 else
245                     break;
246             }
247             if(this._rendererCmd)
248                 cc.renderer.pushRenderCommand(this._rendererCmd);
249             // draw children zOrder >= 0
250             for (; i < childLen; i++) {
251                 if (locChildren[i]) {
252                     locChildren[i].visit();
253                 }
254             }
255         } else{
256             if(this._rendererCmd)
257                 cc.renderer.pushRenderCommand(this._rendererCmd);
258         }
259 
260         cc.renderer.pushRenderCommand(this._afterVisitCmd);
261 
262         //optimize performance for javascript
263         currentStack.top = currentStack.stack.pop();
264     },

266     _onBeforeVisit: function(ctx){
267         var gl = ctx || cc._renderContext;
268         ///////////////////////////////////
269         // INIT
270 
271         // increment the current layer
272         cc.ClippingNode._layer++;
273 
274         // mask of the current layer (ie: for layer 3: 00000100)
275         var mask_layer = 0x1 << cc.ClippingNode._layer;
276         // mask of all layers less than the current (ie: for layer 3: 00000011)
277         var mask_layer_l = mask_layer - 1;
278         // mask of all layers less than or equal to the current (ie: for layer 3: 00000111)
279         //var mask_layer_le = mask_layer | mask_layer_l;
280         this._mask_layer_le = mask_layer | mask_layer_l;
281         // manually save the stencil state
282         this._currentStencilEnabled = gl.isEnabled(gl.STENCIL_TEST);
283         this._currentStencilWriteMask = gl.getParameter(gl.STENCIL_WRITEMASK);
284         this._currentStencilFunc = gl.getParameter(gl.STENCIL_FUNC);
285         this._currentStencilRef = gl.getParameter(gl.STENCIL_REF);
286         this._currentStencilValueMask = gl.getParameter(gl.STENCIL_VALUE_MASK);
287         this._currentStencilFail = gl.getParameter(gl.STENCIL_FAIL);
288         this._currentStencilPassDepthFail = gl.getParameter(gl.STENCIL_PASS_DEPTH_FAIL);
289         this._currentStencilPassDepthPass = gl.getParameter(gl.STENCIL_PASS_DEPTH_PASS);
290 
291         // enable stencil use
292         gl.enable(gl.STENCIL_TEST);
293         // check for OpenGL error while enabling stencil test
294         //cc.checkGLErrorDebug();
295 
296         // all bits on the stencil buffer are readonly, except the current layer bit,
297         // this means that operation like glClear or glStencilOp will be masked with this value
298         gl.stencilMask(mask_layer);
299 
300         // manually save the depth test state
301         //GLboolean currentDepthTestEnabled = GL_TRUE;
302         //currentDepthTestEnabled = glIsEnabled(GL_DEPTH_TEST);
303         //var currentDepthWriteMask = gl.getParameter(gl.DEPTH_WRITEMASK);
304         this._currentDepthWriteMask = gl.getParameter(gl.DEPTH_WRITEMASK);
305         // disable depth test while drawing the stencil
306         //glDisable(GL_DEPTH_TEST);
307         // disable update to the depth buffer while drawing the stencil,
308         // as the stencil is not meant to be rendered in the real scene,
309         // it should never prevent something else to be drawn,
310         // only disabling depth buffer update should do
311         gl.depthMask(false);
312 
313         ///////////////////////////////////
314         // CLEAR STENCIL BUFFER
315 
316         // manually clear the stencil buffer by drawing a fullscreen rectangle on it
317         // setup the stencil test func like this:
318         // for each pixel in the fullscreen rectangle
319         //     never draw it into the frame buffer
320         //     if not in inverted mode: set the current layer value to 0 in the stencil buffer
321         //     if in inverted mode: set the current layer value to 1 in the stencil buffer
322         gl.stencilFunc(gl.NEVER, mask_layer, mask_layer);
323         gl.stencilOp(!this.inverted ? gl.ZERO : gl.REPLACE, gl.KEEP, gl.KEEP);
324 
325         this._drawFullScreenQuadClearStencil();
326 
327         // DRAW CLIPPING STENCIL
328         // setup the stencil test func like this:
329         // for each pixel in the stencil node
330         //     never draw it into the frame buffer
331         //     if not in inverted mode: set the current layer value to 1 in the stencil buffer
332         //     if in inverted mode: set the current layer value to 0 in the stencil buffer
333         gl.stencilFunc(gl.NEVER, mask_layer, mask_layer);
334         gl.stencilOp(!this.inverted ? gl.REPLACE : gl.ZERO, gl.KEEP, gl.KEEP);
335 
336         if (this.alphaThreshold < 1) {            //TODO desktop
337             // since glAlphaTest do not exists in OES, use a shader that writes
338             // pixel only if greater than an alpha threshold
339             var program = cc.shaderCache.programForKey(cc.SHADER_POSITION_TEXTURECOLORALPHATEST);
340             var alphaValueLocation = gl.getUniformLocation(program.getProgram(), cc.UNIFORM_ALPHA_TEST_VALUE_S);
341             // set our alphaThreshold
342             cc.glUseProgram(program.getProgram());
343             program.setUniformLocationWith1f(alphaValueLocation, this.alphaThreshold);
344             // we need to recursively apply this shader to all the nodes in the stencil node
345             // XXX: we should have a way to apply shader to all nodes without having to do this
346             cc.setProgram(this._stencil, program);
347         }
348     },
349 
350     _drawFullScreenQuadClearStencil: function () {
351         // draw a fullscreen solid rectangle to clear the stencil buffer
352         cc.kmGLMatrixMode(cc.KM_GL_PROJECTION);
353         cc.kmGLPushMatrix();
354         cc.kmGLLoadIdentity();
355         cc.kmGLMatrixMode(cc.KM_GL_MODELVIEW);
356         cc.kmGLPushMatrix();
357         cc.kmGLLoadIdentity();
358         cc._drawingUtil.drawSolidRect(cc.p(-1, -1), cc.p(1, 1), cc.color(255, 255, 255, 255));
359         cc.kmGLMatrixMode(cc.KM_GL_PROJECTION);
360         cc.kmGLPopMatrix();
361         cc.kmGLMatrixMode(cc.KM_GL_MODELVIEW);
362         cc.kmGLPopMatrix();
363     },
364 
365     _onAfterDrawStencil: function(ctx){
366         var gl = ctx || cc._renderContext;
367         // restore alpha test state
368         //if (this.alphaThreshold < 1) {
369         // XXX: we need to find a way to restore the shaders of the stencil node and its childs
370         //}
371 
372         // restore the depth test state
373         gl.depthMask(this._currentDepthWriteMask);
374 
375         ///////////////////////////////////
376         // DRAW CONTENT
377 
378         // setup the stencil test func like this:
379         // for each pixel of this node and its childs
380         //     if all layers less than or equals to the current are set to 1 in the stencil buffer
381         //         draw the pixel and keep the current layer in the stencil buffer
382         //     else
383         //         do not draw the pixel but keep the current layer in the stencil buffer
384         gl.stencilFunc(gl.EQUAL, this._mask_layer_le, this._mask_layer_le);
385         gl.stencilOp(gl.KEEP, gl.KEEP, gl.KEEP);
386     },
387 
388     _onAfterVisit: function(ctx){
389         var gl = ctx || cc._renderContext;
390         ///////////////////////////////////
391         // CLEANUP
392 
393         // manually restore the stencil state
394         gl.stencilFunc(this._currentStencilFunc, this._currentStencilRef, this._currentStencilValueMask);
395         gl.stencilOp(this._currentStencilFail, this._currentStencilPassDepthFail, this._currentStencilPassDepthPass);
396         gl.stencilMask(this._currentStencilWriteMask);
397         if (!this._currentStencilEnabled)
398             gl.disable(gl.STENCIL_TEST);
399 
400         // we are done using this layer, decrement
401         cc.ClippingNode._layer--;
402     },

404     _visitForCanvas: function (ctx) {
405         // Composition mode, costy but support texture stencil
406         this._clipElemType = (this._cangodhelpme() || this._stencil instanceof cc.Sprite);
407 
408         var context = ctx || cc._renderContext;
409         var i, children = this._children, locChild;
410 
411         if (!this._stencil || !this._stencil.visible) {
412             if (this.inverted)
413                 cc.Node.prototype.visit.call(this, ctx);   // draw everything
414             return;
415         }
416 
417         if(this._rendererSaveCmd)
418             cc.renderer.pushRenderCommand(this._rendererSaveCmd);
419 
420         if(this._clipElemType){
421 
422             // Draw everything first using node visit function
423             cc.Node.prototype.visit.call(this, context);
424         }else{
425             this._stencil.visit(context);
426         }
427 
428         if(this._rendererClipCmd)
429             cc.renderer.pushRenderCommand(this._rendererClipCmd);
430 
431         this.transform();
432 
433         if(this._clipElemType){
434             this._stencil.visit();
435         }else{
436             // Clip mode doesn't support recusive stencil, so once we used a clip stencil,
437             // so if it has ClippingNode as a child, the child must uses composition stencil.
438             this._cangodhelpme(true);
439             var len = children.length;
440             if (len > 0) {
441                 this.sortAllChildren();
442                 // draw children zOrder < 0
443                 for (i = 0; i < len; i++) {
444                     locChild = children[i];
445                     if (locChild._localZOrder < 0)
446                         locChild.visit(context);
447                     else
448                         break;
449                 }
450                 if(this._rendererCmd)
451                     cc.renderer.pushRenderCommand(this._rendererCmd);
452                 for (; i < len; i++) {
453                     children[i].visit(context);
454                 }
455             } else
456             if(this._rendererCmd)
457                 cc.renderer.pushRenderCommand(this._rendererCmd);
458             this._cangodhelpme(false);
459 
460         }
461 
462         if(this._rendererRestoreCmd)
463             cc.renderer.pushRenderCommand(this._rendererRestoreCmd);
464     },

492     _setStencilForCanvas: function (stencil) {
493         this._stencil = stencil;
494         if(stencil._buffer){
495             for(var i=0; i<stencil._buffer.length; i++){
496                 stencil._buffer[i].isFill = false;
497                 stencil._buffer[i].isStroke = false;
498             }
499         }
500         var locContext = cc._renderContext;
501         // For texture stencil, use the sprite itself
502         //if (stencil instanceof cc.Sprite) {
503         //    return;
504         //}
505         // For shape stencil, rewrite the draw of stencil ,only init the clip path and draw nothing.
506         //else
507         if (stencil instanceof cc.DrawNode) {
508             stencil._rendererCmd.rendering = function (ctx, scaleX, scaleY) {
509                 scaleX = scaleX || cc.view.getScaleX();
510                 scaleY = scaleY ||cc.view.getScaleY();
511                 var context = ctx || cc._renderContext;
512                 var t = this._node._transformWorld;
513                 context.save();
514                 context.transform(t.a, t.b, t.c, t.d, t.tx * scaleX, -t.ty * scaleY);
515 
516                 context.beginPath();
517                 for (var i = 0; i < stencil._buffer.length; i++) {
518                     var vertices = stencil._buffer[i].verts;
519                     //cc.assert(cc.vertexListIsClockwise(vertices),
520                     //    "Only clockwise polygons should be used as stencil");
521 
522                     var firstPoint = vertices[0];
523                     context.moveTo(firstPoint.x * scaleX, -firstPoint.y * scaleY);
524                     for (var j = 1, len = vertices.length; j < len; j++)
525                         context.lineTo(vertices[j].x * scaleX, -vertices[j].y * scaleY);
526                 }
527                 context.restore();
528             };
529         }
